  cloud_agentlib
  STATIC
    entrypoint.cc
    collector.cc
    enumerator.cc
    ingest_connection.cc
)

  cloud_agentlib
  PRIVATE
    ${PROJECT_SOURCE_DIR}
    ${CMAKE_CURRENT_BINARY_DIR}
)

  cloud_agentlib
  PUBLIC
    signal_handler 
)
